# For Fresh Install of Ubuntu

```
sudo apt-get update
sudo apt-get upgrade


# Chromium
sudo apt-get install chromium-browser

# Git
sudo apt-get install git

# atom
wget https://github.com/atom/atom/releases/download/v1.19.3/atom-amd64.deb
sudo dpkg --install atom-amd64.deb



# Install Java & Eclipse
sudo apt-get install default-jdk
sudo update-alternatives --config java
sudo nano /etc/environment   # replace JAVA_HOME="/usr/lib/jvm/java-8-oracle"
source /etc/environment

# Install Anaconda - download the installation file
bash Anaconda.....




```
